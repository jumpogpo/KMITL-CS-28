```python
def bottle_neck_path(g, distances):
    for k in range(1, g.V+1):
        for i in range(1, g.V+1):
            for j in range(1, g.V+1):
                tmp = min(distances[i][j], max(distances[i][k], distances[k][j]))
                distances[i][j] = tmp
                distances[j][i] = tmp
    return distances
```
